88 Point3D Point3D::rotate(double t, const Point3D& a) const
89 {
90  t /= 2.0;
91  const double sn = sin(t), ro = cos(t);
92  // Euler parameters: lam, mu, nu
93  const double lam = a.x*sn, mu = a.y*sn, nu = a.z*sn,
94  LamMu = lam*mu, LamNu = lam*nu, LamRo = lam*ro,
95  RoMu = ro*mu, RoNu = ro*nu, MuNu = mu*nu,
96  sqr_lam = SQR(lam), sqr_mu = SQR(mu),
97  sqr_nu = SQR(nu), sqr_ro = SQR(ro);
98  // elements of the rotation matrix a[3X3]
99  const double a11 = sqr_lam-sqr_mu-sqr_nu+sqr_ro,
100  a12 = LamMu-RoNu,
101  a13 = LamNu+RoMu,
102  a21 = LamMu+RoNu,
103  a22 = sqr_mu-sqr_nu-sqr_lam+sqr_ro,
104  a23 = MuNu-LamRo,
105  a31 = LamNu-RoMu,
106  a32 = MuNu+LamRo,
107  a33 = sqr_nu-sqr_lam-sqr_mu+sqr_ro;
108  return Point3D(this->x*a11 + 2.0*(this->y*a12+this->z*a13),
109  this->y*a22 + 2.0*(this->x*a21+this->z*a23),
110  this->z*a33 + 2.0*(this->x*a31+this->y*a32));
111 }
